def rk4_fixed_step(rhs, y0, t0, dt, n_steps):
    t = t0
    y = y0
    for i in range(n_steps):
        k1 = dt * rhs(t, y)
        k2 = dt * rhs(t + dt/2, y + k1/2)
        k3 = dt * rhs(t + dt/2, y + k2/2)
        k4 = dt * rhs(t + dt, y + k3)
        y = y + (k1 + 2*k2 + 2*k3 + k4) / 6
        t += dt
    return y
